\section{Contexte du Projet}

% ────────────────────────────────────────────────────────────
%  SLIDE 3 — CONTEXTE
% ────────────────────────────────────────────────────────────
\begin{frame}{Contexte du Projet}
  \framedesc{Présentation du problème existant et des enjeux qui justifient la réalisation de ce projet.}

  \begin{columns}[T]

    % Colonne gauche — Description du problème
    \begin{column}{0.52\textwidth}
      \begin{block}{Problématique identifiée}
        La gestion des matériels de transmission dans les unités militaires
        repose encore sur des \textbf{registres papier} et des fichiers
        Excel non centralisés, entraînant :
        \begin{itemize}
          \item Perte de données et erreurs de saisie
          \item Absence de traçabilité des mouvements
          \item Impossible suivi en temps réel des stocks
          \item Processus lents et redondants
        \end{itemize}
      \end{block}
    \end{column}

    % Colonne droite — Importance
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{Enjeux}
        \begin{itemize}
          \item \textbf{Opérationnel} : disponibilité du matériel critique
          \item \textbf{Sécurité} : traçabilité rigoureuse
          \item \textbf{Efficacité} : réduction des délais administratifs
          \item \textbf{Modernisation} : transformation numérique
        \end{itemize}
      \end{alertblock}

      \vspace{0.3cm}

      \begin{exampleblock}{Contexte}
        Projet de stage initié en \textbf{Janvier 2026} dans le cadre
        de la modernisation du système d'information de la
        Direction des Transmissions.
      \end{exampleblock}
    \end{column}

  \end{columns}
\end{frame}

% ============================================================
%  SECTION 2 — OBJECTIFS
% ============================================================
\section{Objectifs du Projet}

% ────────────────────────────────────────────────────────────
%  SLIDE 4 — OBJECTIFS
% ────────────────────────────────────────────────────────────
\begin{frame}{Objectifs du Projet}
  \framedesc{Définition des cibles à atteindre à l'issue du développement, du général au spécifique.}

  \begin{block}{Objectif Général}
    Développer une \textbf{application web} centralisée permettant la
    gestion complète des matériels de transmission : inventaire,
    mouvements, traçabilité et reporting.
  \end{block}

  \vspace{0.4cm}

  \textbf{Objectifs Spécifiques :}

  \vspace{0.2cm}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{enumerate}
        \item Concevoir un modèle de données adapté aux besoins militaires
        \item Développer un module de gestion des matériels (CRUD)
        \item Implémenter un système de traçabilité des mouvements
        \item Créer un tableau de bord analytique en temps réel
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{enumerate}
        \setcounter{enumi}{4}
        \item Gérer les utilisateurs avec contrôle d'accès par rôles
        \item Générer des rapports et statistiques exportables
        \item Assurer l'authentification sécurisée
        \item Déployer sur un serveur accessible au sein du réseau
      \end{enumerate}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Analyse des Besoins}
  \framedesc{Identification et classification des exigences fonctionnelles, non fonctionnelles et des contraintes du système.}

  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      \begin{block}{Besoins Fonctionnels}
        \begin{itemize}
          \item Gestion des matériels (ajout, modification, suppression)
          \item Enregistrement des mouvements (entrée/sortie)
          \item Suivi des détenteurs et responsables
          \item Tableau de bord avec statistiques
          \item Gestion documentaire (ordres 50/4, 50/5…)
          \item Authentification et gestion des rôles
        \end{itemize}
      \end{block}
    \end{column}

    \begin{column}{0.48\textwidth}
      \begin{block}{Besoins Non Fonctionnels}
        \begin{itemize}
          \item \textbf{Sécurité} : authentification, CSRF, RBAC
          \item \textbf{Performance} : chargement $<$ 2 secondes
          \item \textbf{Accessibilité} : interface responsive (PC, tablette)
          \item \textbf{Fiabilité} : sauvegarde automatique des données
          \item \textbf{Maintenabilité} : code documenté, modulaire
        \end{itemize}
      \end{block}

      \vspace{0.2cm}

      \begin{alertblock}{Contraintes}
        Réseau interne fermé, pas d'accès Internet,\\
        compatibilité navigateurs anciens requise.
      \end{alertblock}

    \end{column}

  \end{columns}
